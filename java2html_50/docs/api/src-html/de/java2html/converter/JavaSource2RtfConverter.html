001 package de.java2html.converter;
002 
003 import java.io.BufferedWriter;
004 import java.io.IOException;
005 import java.io.Writer;
006 
007 import de.java2html.javasource.JavaSource;
008 import de.java2html.javasource.JavaSourceIterator;
009 import de.java2html.javasource.JavaSourceRun;
010 import de.java2html.javasource.JavaSourceType;

016 /**
017  * @author Markus Gebhard
018  */
019 public class JavaSource2RtfConverter extends AbstractJavaSourceConverter {
020 
021   public JavaSource2RtfConverter() {
022     super(new ConverterMetaData("rtf", "RTF (Rich Text Format)", "rtf"));
023   }
024 
025   private final static int FONT_SIZE = 9;
026   private final static String FONT_NAME = "Courier New";
027 
028   public String getDocumentHeader(JavaSourceConversionOptions options, String title) {
029     return "";
030   }
031 
032   public String getDocumentFooter(JavaSourceConversionOptions options) {
033     return "";
034   }
035 
036   public String getBlockSeparator(JavaSourceConversionOptions options) {
037     return "";
038   }
039 
040   public void convert(JavaSource source, JavaSourceConversionOptions options, BufferedWriter writer)
041       throws IOException {
042     writer.write("{\\rtf1\\ansi\\deff0{\\fonttbl{");
043     writer.write("\\f0\\fmodern ");
044     writer.write(FONT_NAME);
045     writer.write(";");
046     writer.write("}");
047     writer.write("}\n");
048 
049     writer.write("{\\colortbl");
050     JavaSourceType[] types = JavaSourceType.getAll();
051     for (int i = 0; i < types.length; ++i) {
052       JavaSourceStyleEntry style = options.getStyleTable().get(types[i]);
053       writeColor(style.getColor(), writer);
054     }
055     writer.write("}\n");
056 
057     writer.write("\\deflang1031\\pard\\plain\\f0");
058     writer.write("\\fs" + FONT_SIZE * 2);
059     writer.write("\\cb" + JavaSourceType.BACKGROUND.getID());
060 
061     int lineCifferCount = String.valueOf(source.getLineCount()).length();
062     JavaSourceIterator iterator = source.getIterator();
063     int lineNumber = 1;
064     while (iterator.hasNext()) {
065       JavaSourceRun run = iterator.getNext();
066 
067       if (run.isAtStartOfLine()) {
068         if (lineNumber > 1) {
069           writer.newLine();
070           writer.write("\\par ");
071         }
072 
073         if (options.isShowLineNumbers()) {
074           writeLineNumber(writer, lineCifferCount, lineNumber, options);
075         }
076         ++lineNumber;
077       }
078       writeText(options, run, writer);
079     }
080 
081     writer.write("}");
082   }
083 
084   private void writeText(JavaSourceConversionOptions options, JavaSourceRun run, BufferedWriter writer)
085       throws IOException {
086     writeText(options.getStyleTable(), writer, run.getCode(), run.getType());
087   }
088 
089   private void writeText(JavaSourceStyleTable styleTable, BufferedWriter writer, String text, JavaSourceType type)
090       throws IOException {
091     JavaSourceStyleEntry style = styleTable.get(type);
092 
093     writer.write("{\\f0");
094     if (style.isBold()) {
095       writer.write("\\b");
096     }
097     if (style.isItalic()) {
098       writer.write("\\i");
099     }
100 
101     writer.write("\\cf" + type.getID() + " ");
102 
103     for (int i = 0; i < text.length(); ++i) {
104       char ch = text.charAt(i);
105       if (ch == '\\') {
106         writer.write("\\\\");
107       }
108       else if (ch == '{') {
109         writer.write("\\{");
110       }
111       else if (ch == '}') {
112         writer.write("\\}");
113       }
114       else {
115         writer.write(ch);
116       }
117     }
118 
119     writer.write("}");
120   }
121 
122   private void writeLineNumber(
123       BufferedWriter writer,
124       int lineCifferCount,
125       int lineNumber,
126       JavaSourceConversionOptions options) throws IOException {
127 
128     String text = String.valueOf(lineNumber);
129     int cifferCount = lineCifferCount - text.length();
130     while (cifferCount > 0) {
131       text = '0' + text;
132       --cifferCount;
133     }
134     text = text + " ";
135 
136     writeText(options.getStyleTable(), writer, text, JavaSourceType.LINE_NUMBERS);
137   }
138 
139   private static void writeColor(RGB color, Writer writer) throws IOException {
140     writer.write("\\red" + color.getRed()); //$NON-NLS-1$
141     writer.write("\\green" + color.getGreen()); //$NON-NLS-1$
142     writer.write("\\blue" + color.getBlue()); //$NON-NLS-1$
143     writer.write(";"); //$NON-NLS-1$
144   }
