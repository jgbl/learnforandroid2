003 import java.io.BufferedWriter;
004 import java.io.IOException;
005 
006 import de.java2html.javasource.JavaSourceType;

011 /**
012  * @author Markus Gebhard
013  */
014 public abstract class AbstractJavaSourceToXmlConverter extends AbstractJavaSourceConverter {
015 
016   //original style sheet by Jan Tisje
017   // "td.java, td.java-ln {vertical-align:top;}\n" +
018   // "tt.java, tt.java-ln, pre.java, pre.java-ln {line-height:1em; margin-bottom:0em;}\n" +
019   // "td.java-ln { text-align:right; }\n"+
020   // "tt.java-ln, pre.java-ln { color:#888888 }\n"+
021   // "/* UNDEFINED       */ span.java0  { color:black; }\n" +
022   // "/* CODE            */ span.java1  { color:black; }\n" +
023   // "/* CODE_KEYWORD    */ span.java2  { color:black; font-weight:bold; }\n" +
024   // "/* CODE_TYPE       */ span.java3  { color:darkblue ;}\n" +
025   // "/* QUOTE           */ span.java4  { color:darkgreen; }\n" +
026   // "/* COMMENT_LINE    */ span.java5  { color:#888888; font-style:italic; }\n" +
027   // "/* COMMENT_BLOCK   */ span.java6  { color:#888888; font-style:italic; }\n" +
028   // "/* COMMENT_JAVADOC */ span.java7  { color:green; font-style:italic; }\n" +
029   // "/* COMMENT_KEYWORD */ span.java8  { color:green; font-style:italic; font-weight:bold; }\n" +
030   // "/* EMPTY           */ span.java9  {}\n" +
031   // "/* NUM_CONSTANT    */ span.java10 { color:orange; }\n" +
032   // "/* CHAR_CONSTANT   */ span.java11 { color:red; } ";
033 
034   protected final static String XML_HEADER = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n";
035 
036   protected boolean lineNumbers = true;
037   protected boolean pre = true;
038   protected String lineEnd = "";
039   private String space = " ";
040   private String headEnd = "";
041   private String foot = "";
042 
043   private final static String TAG_START = "<span class=\"java";
044   private final static String TAG_END = "\">";
045   private final static String TAG_CLOSE = "</span>";
046 
047   public AbstractJavaSourceToXmlConverter(ConverterMetaData metaData) {
048     super(metaData);
049     setOptions(false, false);
050   }
051 
052   /** @deprecated As of Sep 13, 2004 (Markus Gebhard) Only options in the 
053    * Java2HtmlConversionOptions are available */
054   public void setOptions(boolean lineNumbers, boolean pre) {
055     this.lineNumbers = lineNumbers;
056     this.pre = pre;
057     foot = getFooter();
058     headEnd = getHeaderEnd();
059     if (pre) {
060       lineEnd = "";
061       space = " "; // normal space
062     }
063     else {
064       lineEnd = "<br />";
065       space = "&#xA0;";
066     }
067   }
068 
069   public String getDocumentHeader(JavaSourceConversionOptions options, String title) {
070     return createHeader(options.getStyleTable(), title) + headEnd;
071   }
072 
073   public String getDocumentFooter(JavaSourceConversionOptions options) {
074     return foot;
075   }
076 
077   protected abstract String createHeader(JavaSourceStyleTable styleTable, String title);
078 
079   protected abstract String getHeaderEnd();
080 
081   protected abstract String getFooter();
082 
083   protected static String createStyleSheet(JavaSourceStyleTable styleTable) {
084     StringBuffer buffer = new StringBuffer();
085     buffer.append("td.java, td.java-ln {vertical-align:top;}\n"
086         + "tt.java, tt.java-ln, pre.java, pre.java-ln {line-height:1em; margin-bottom:0em;}\n"
087         + "td.java-ln { text-align:right; }\n"
088         + "tt.java-ln, pre.java-ln { color:#888888 }\n");
089 
090     JavaSourceType[] types = JavaSourceType.getAll();
091     for (int i = 0; i < types.length; i++) {
092       appendStyle(buffer, types[i], styleTable);
093     }
094     return buffer.toString();
095   }
096 
097   private static void appendStyle(StringBuffer buffer, JavaSourceType type, JavaSourceStyleTable styleTable) {
098     buffer.append("/* " + type.getName() + "       */ ");
099     buffer.append("span.java" + type.getID() + "  { ");
100     buffer.append("font-size: 10pt; ");
101     buffer.append("color:" + styleTable.get(type).getHtmlColor() + "; ");
102     if (styleTable.get(type).isBold()) {
103       buffer.append("font-weight:bold; ");
104     }
105     if (styleTable.get(type).isItalic()) {
106       buffer.append("font-style:italic; ");
107     }
108     buffer.append("}\n");
109   }
110 
111   public String getBlockSeparator(JavaSourceConversionOptions options) {
112     return "";
113   }
114 
115   protected void toXml(String sourceCode, JavaSourceType[] sourceTypes, int start, int end, BufferedWriter writer)
116       throws IOException {
117     writer.write(TAG_START + sourceTypes[start].getID() + TAG_END);
118 
119     String t = HtmlUtilities.encode(sourceCode, start, end + 1, "\r\n ");
120 
121     for (int i = 0; i < t.length(); ++i) {
122       char ch = t.charAt(i);
123       if (ch == ' ')
124         if ((i < t.length() - 1) && (t.charAt(i + 1) == ' ')) {
125           writer.write(space);
126         }
127         else {
128           writer.write(" ");
129         }
130       else if (ch == '\n') {
131         writer.write(lineEnd + "\n");
132       }
133       else if (ch != '\r') {
134         writer.write(ch);
135       }
136     }
137 
138     writer.write(TAG_CLOSE);
139   }
