047 public class InfoDialog extends Dialog implements ActionListener, KeyListener, WindowListener {
048 
049   protected final static Font FONT_HEADLINE = new Font("Dialog", Font.BOLD, 13);
050 
051   public InfoDialog(Frame parent, String title, Component content) {
052     this(parent, title, "", content);
053   }
054 
055   public InfoDialog(Frame parent, String title, String headline, Component content) {
056     super(parent, title, true);
057     init(headline, content);
058   }
059 
060   public InfoDialog(Frame parent, String title, String text) {
061     this(parent, title, "", text);
062   }
063 
064   public InfoDialog(Frame parent, String title, String headline, String text) {
065     super(parent, title, true);
066 
067     GBorderedPanel content = new GBorderedPanel();
068     content.setLayout(new FlowLayout(FlowLayout.CENTER, 5, 0));
069     content.add(new GLabel(text, GLabel.LEFT)); //Text selbst nach links
070     // ausgerichtet
071 
072     init(headline, content);
073   }
074 
075   public void keyTyped(KeyEvent evt) {
076     //nothing to do
077   }
078   public void keyReleased(KeyEvent evt) {
079     //nothing to do
080   }
081   public void keyPressed(KeyEvent evt) {
082     if (evt.getKeyCode() == KeyEvent.VK_ESCAPE) {
083       dispose();
084     }
085   }
086 
087   protected void init(String headline, Component content) {
088     addWindowListener(this);
089     addKeyListener(this);
090 
091     setLayout(new BorderLayout());
092 
093     //Header - if any
094     if (headline != null && !headline.equals("")) {
095       Label lh = new Label(headline, Label.CENTER);
096       lh.setFont(FONT_HEADLINE);
097       add(lh, BorderLayout.NORTH);
098     }
099 
100     add(content, BorderLayout.CENTER);
101 
102     //OK-Button
103     Button bOk = new Button("    Ok    ");
104     bOk.addActionListener(this);
105     Panel p2 = new Panel();
106     p2.setLayout(new FlowLayout(FlowLayout.RIGHT));
107     p2.add(bOk);
108     add(p2, BorderLayout.SOUTH);
109 
110     pack();
111     setResizable(false);
112     GuiTools.centerOnScreen(this);
113   }
114 
115   public void actionPerformed(ActionEvent evt) {
116     dispose();
117   }
118 
119   //WindowListener
120   public void windowOpened(WindowEvent e) {
121     //nothing to do
122   }
123   public void windowClosing(WindowEvent e) {
124     dispose();
125   }
126   public void windowClosed(WindowEvent e) {
127     //nothing to do
128   }
129   public void windowIconified(WindowEvent e) {
130     //nothing to do
131   }
132   public void windowDeiconified(WindowEvent e) {
133     //nothing to do
134   }
135   public void windowActivated(WindowEvent e) {
136     //nothing to do
137   }
138   public void windowDeactivated(WindowEvent e) {
139     //nothing to do
140   }
