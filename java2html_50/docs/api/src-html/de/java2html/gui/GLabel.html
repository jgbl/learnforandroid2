041 public class GLabel extends Canvas {
042   protected String[] lines;
043   protected int lineHeight;
044   protected int blockWidth;
045   protected int blockHeight;
046   protected int fontAscent;
047 
048   /**
049    * Indicates that the label should be left justified.
050    */
051   public static final int LEFT = 0;
052 
053   /**
054    * Indicates that the label should be centered.
055    */
056   public static final int CENTER = 1;
057 
058   /**
059    * Indicates that the label should be right justified.
060    * @since   JDK1.0t.
061    */
062   public static final int RIGHT = 2;
063 
064   /**
065    * The label's alignment.  The default alignment is set
066    * to be left justified.
067    */
068   int alignment = LEFT;
069 
070   public GLabel(String text) {
071     this(text, LEFT);
072   }
073 
074   public GLabel(String text, int alignment) {
075     this.alignment = alignment;
076 
077     setText(text);
078   }
079 
080   public void setText(String text) {
081     if (text == null)
082       text = "";
083 
084     //Alle Tabs durch mehrere Leerzeichen ersetzen
085     StringBuffer sb = new StringBuffer();
086     StringTokenizer st = new StringTokenizer(text, "\t", true);
087     while (st.hasMoreTokens()) {
088       String s = st.nextToken();
089       if (s.charAt(0) == '\t')
090         sb.append("      ");
091       else
092         sb.append(s);
093     }
094     text = sb.toString();
095 
096     //Anzahl an Zeilen (=Anzahl von '\n'+1) zählen
097     int l = 1;
098     int i1 = text.indexOf('\n');
099     while (i1 != -1) {
100       ++l;
101       i1 = text.indexOf('\n', i1 + 1);
102     }
103 
104     lines = new String[l];
105 
106     int lineIndex = 0;
107     i1 = 0;
108     int i2 = text.indexOf('\n');
109     while (i2 != -1) {
110       lines[lineIndex++] = text.substring(i1, i2);
111       i1 = i2 + 1;
112       i2 = text.indexOf('\n', i1);
113     }
114 
115     //final line
116     lines[lineIndex] = text.substring(i1);
117 
118     //Invalidieren! => getPreferredSize liefert neue werte
119     blockWidth = 0;
120     invalidate();
121 
122     Component parent = getParent();
123     if (parent != null)
124       parent.validate();
125 
126     //
127     repaint();
128   }
129 
130   public Dimension getPreferredSize() {
131     if (blockWidth != 0 && blockHeight != 0)
132       return new Dimension(blockWidth, blockHeight);
133 
134     FontMetrics fm = getFontMetrics(getFont());
135     blockWidth = 0;
136     blockHeight = 0;
137 
138     for (int i = 0; i < lines.length; ++i) {
139       int l = fm.stringWidth(lines[i]);
140       if (l > blockWidth)
141         blockWidth = l;
142     }
143     fontAscent = fm.getAscent();
144     lineHeight = fm.getHeight();
145     blockHeight = lines.length * lineHeight;
146 
147     return new Dimension(blockWidth, blockHeight);
148   }
149 
150   public Dimension getMinimumSize() {
151     return getPreferredSize();
152   }
153 
154   public Dimension getMaximumSize() {
155     return getPreferredSize();
156   }
157 
158   public void paint(Graphics g) {
159     update(g);
160   }
161 
162   public void update(Graphics g) {
163     Dimension d = getSize();
164     g.setColor(getBackground());
165     g.fillRect(0, 0, d.width, d.height);
166 
167     if (isEnabled()) {
168       g.setColor(SystemColor.controlText);
169       draw(g, 0, 0);
170     } else {
171       g.setColor(SystemColor.controlLtHighlight);
172       draw(g, 1, 1);
173       g.setColor(SystemColor.controlShadow);
174       draw(g, 0, 0);
175     }
176   }
177 
178   protected void draw(Graphics g, int x0, int y0) {
179     if (alignment == LEFT) {
180       for (int i = 0; i < lines.length; ++i) {
181         g.drawString(lines[i], x0, y0 + fontAscent + i * lineHeight);
182       }
183     } else if (alignment == RIGHT) {
184       Dimension d = getSize();
185       FontMetrics fm = getFontMetrics(getFont());
186       for (int i = 0; i < lines.length; ++i) {
187         int l = fm.stringWidth(lines[i]);
188         g.drawString(lines[i], x0 + d.width - l, y0 + fontAscent + i * lineHeight);
189       }
190     } else if (alignment == CENTER) {
191       Dimension d = getSize();
192       FontMetrics fm = getFontMetrics(getFont());
193       for (int i = 0; i < lines.length; ++i) {
194         int l = fm.stringWidth(lines[i]);
195         g.drawString(lines[i], x0 + (d.width - l) / 2, y0 + fontAscent + i * lineHeight);
196       }
197     }
198   }
