001 package de.java2html.plugin.jspwiki.test;
002 
003 import java.util.HashMap;
004 
005 import junit.framework.TestCase;
006 
007 import com.ecyrd.jspwiki.plugin.PluginException;
008 
009 import de.java2html.JavaSourceConversionSettings;
010 import de.java2html.options.JavaSourceStyleTable;
011 import de.java2html.plugin.AbstractJava2HtmlPlugin;
012 import de.java2html.plugin.jspwiki.Java2HtmlPlugin;
013 
014 /**
015  * @author Markus Gebhard
016  */
017 public class Java2HtmlPluginTest extends TestCase {
018   private final static String SOURCE_CODE = "public class HelloWorld {\n"
019       + "  public static void main(String args[]) {\n"
020       + "    System.out.println(\"Hello World!\");\n"
021       + "  }\n"
022       + "}";
023 
024   private final static String CONVERTED_HTML_DEFAULT_STYLE = AbstractJava2HtmlPlugin.convert(
025       SOURCE_CODE,
026       AbstractJava2HtmlPlugin.getDefaultSettings());
027 
028   private Java2HtmlPlugin plugin;
029 
030   private static String CONVERTED_HTML_MONOCHROME_STYLE;
031 
032   static {
033     JavaSourceConversionSettings monochromeOptions = AbstractJava2HtmlPlugin.getDefaultSettings();
034     monochromeOptions.getConversionOptions().setStyleTable(JavaSourceStyleTable.getDefaultMonochromeStyleTable());
035     CONVERTED_HTML_MONOCHROME_STYLE = AbstractJava2HtmlPlugin.convert(SOURCE_CODE, monochromeOptions);
036   }
037 
038   protected void setUp() throws Exception {
039     plugin = new Java2HtmlPlugin();
040 
041   }
042 
043   public void testThrowsUsageExceptionWhenNoSourceCodeGiven() {
044     try {
045       plugin.execute(null, new HashMap());
046       fail();
047     }
048     catch (PluginException expected) {
049       assertEquals(Java2HtmlPlugin.DEFAULT_USAGE_MESSAGE, expected.getMessage());
050     }
051   }
052 
053   public void testConversionAsSourceParameter() throws PluginException {
054     HashMap map = new HashMap();
055     map.put("source", SOURCE_CODE);
056     assertEquals(CONVERTED_HTML_DEFAULT_STYLE, plugin.execute(null, map));
057   }
058 
059   public void testConversionAsBodyParameter() throws PluginException {
060     HashMap map = new HashMap();
061     map.put("_body", SOURCE_CODE);
062     assertEquals(CONVERTED_HTML_DEFAULT_STYLE, plugin.execute(null, map));
063   }
064 
065   public void testConversionAsBodyParameterWithLeadingNewLine() throws PluginException {
066     /*
067      * Reason: JSPWiki does not automatically remove the extra newline in the
068      * multiline parameter _body
069      */
070     HashMap map = new HashMap();
071     map.put("_body", "\n" + SOURCE_CODE);
072     assertEquals(CONVERTED_HTML_DEFAULT_STYLE, plugin.execute(null, map));
073   }

087   public void testUsingDefaultStyleNameIsSameAsUsingDefaultStyle() throws PluginException {
088     HashMap map = new HashMap();
089     map.put("source", SOURCE_CODE);
090     map.put("style", AbstractJava2HtmlPlugin
091         .getDefaultSettings().getConversionOptions().getStyleTable().getName());
092     assertEquals(CONVERTED_HTML_DEFAULT_STYLE, plugin.execute(null, map));
093   }
094 
095   public void testUsingMonochromeStyle() throws PluginException {
096     HashMap map = new HashMap();
097     map.put("source", SOURCE_CODE);
098     map.put("style", JavaSourceStyleTable.getDefaultMonochromeStyleTable().getName());
099     assertEquals(CONVERTED_HTML_MONOCHROME_STYLE, plugin.execute(null, map));
100   }

102   public void testUnsupportedConversionStyle() {

105     try {
106       plugin.execute(null, map);
107     }

109       //expected
