08 public abstract class BasicTestCase extends TestCase {
09   public static void assertThrowsException(Class expectedExceptionClass, IBlock block) {
10     try {
11       block.execute();
12       fail("Exception of type " + expectedExceptionClass + " expected.");  //$NON-NLS-1$//$NON-NLS-2$
13     }
14     catch (Exception exception) {
15       assertTrue("Exception of type " //$NON-NLS-1$
16           + expectedExceptionClass
17           + " expecrted, but was " //$NON-NLS-1$
18           + exception.getClass(), expectedExceptionClass.isAssignableFrom(exception.getClass()));
19     }
20   }
21 
22   public static void assertInstanceOf(Class expectedClass, Object actualObject) {
23     assertNotNull(actualObject);
24     assertTrue(expectedClass.isAssignableFrom(actualObject.getClass()));
25   }
26 
27   public static interface IBlock {
28     public void execute() throws Exception;
29   }
