080   }

082   private static JavaSourceConversionOptions createDefaultOptions() throws IllegalConfigurationException {
083     InputStream inputStream = JavaSourceConversionOptions.class.getClassLoader().getResourceAsStream(
084         PROPERTIES_FILE_NAME);
085     if (inputStream == null) {
086       return new JavaSourceConversionOptions();
087     }
088 
089     /* TODO: ClassLoader.getSystemResourceAsStream(PROPERTIES_FILE_NAME)
090      instead of
091      Java2HtmlConversionOptions.class.getClassLoader().getResourceAsStream(PROPERTIES_FILE_NAME);
092      Will have to find out how I can make this compatible when using Java WebStart.
093      As far as I know using ClassLoader.getSystemResourceAsStream() will propbably not find 
094      resources that are defined in the JNLP file. Maybe I should use 
095      ClassLoader.getSystemResourceAsStream as fallback if there is no file found using the other
096      classloader... */
097 
098     Properties properties = new Properties();
099     try {
100       properties.load(inputStream);
101       return new ConversionOptionsPropertiesReader().read(properties);
102     }
103     catch (IOException exception) {
104       throw new IllegalConfigurationException("Error loading configuration file '" //$NON-NLS-1$
105           + PROPERTIES_FILE_NAME + "' from classpath", exception); //$NON-NLS-1$
106     }
107     catch (IllegalArgumentException exception) {
108       throw new IllegalConfigurationException("Error loading configuration file '" //$NON-NLS-1$
109           + PROPERTIES_FILE_NAME + "' from classpath", exception); //$NON-NLS-1$
110     }
111     finally {
112       IoUtilities.close(inputStream);
113     }
114   }

135   public boolean equals(Object obj) {
136     if (!(obj instanceof JavaSourceConversionOptions)) {
137       return false;
138     }
139     JavaSourceConversionOptions other = (JavaSourceConversionOptions) obj;
140     return (other.tabSize == tabSize
141         && other.styleTable.equals(styleTable)
142         && other.showFileName == showFileName
143         && other.showJava2HtmlLink == showJava2HtmlLink
144         && other.showLineNumbers == showLineNumbers
145         && other.showTableBorder == showTableBorder && other.horizontalAlignment == horizontalAlignment);
146   }
