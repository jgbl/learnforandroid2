010 /**
011  * Table containing style options (
012  * {@link de.java2html.options.JavaSourceStyleEntry}) for different types of
013  * source code ({@link de.java2html.javasource.JavaSourceType}).
014  * 
015  * @author Markus Gebhard
016  */
017 public class JavaSourceStyleTable {
018   private final Map table = new HashMap();
019   private static JavaSourceStyleEntry defaultColorEntry = new JavaSourceStyleEntry(RGB.BLACK);
020   private String name;
021 
022   private JavaSourceStyleTable(String name) {
023     setName(name);
024   }
025 
026   public boolean equals(Object obj) {
027     if (!(obj instanceof JavaSourceStyleTable)) {
028       return false;
029     }
030     JavaSourceStyleTable other = (JavaSourceStyleTable) obj;
031 
032     if (!name.equals(other.name)) {
033       return false;
034     }
035     if (other.table.size() != table.size()) {
036       return false;
037     }
038 
039     Iterator iterator = table.keySet().iterator();
040     while (iterator.hasNext()) {
041       String key = (String) iterator.next();
042       JavaSourceStyleEntry value = get(key);
043       if (!value.equals(other.table.get(key))) {
044         return false;
045       }
046     }
047     return true;
048   }
049 
050   public JavaSourceStyleTable getClone() {
051     JavaSourceStyleTable clone = new JavaSourceStyleTable(getName());
052     Iterator iterator = table.keySet().iterator();
053     while (iterator.hasNext()) {
054       String key = (String) iterator.next();
055       JavaSourceStyleEntry value = get(key);
056       clone.table.put(key, value);
057     }
058     return clone;
059   }

061   public static JavaSourceStyleTable[] getPredefinedTables() {
062     return new JavaSourceStyleTable[]{
063         createDefaultEclipseStyleTable(),
064         createDefaultKawaStyleTable(),
065         createDefaultMonochromeStyleTable(), };
066   }

068   /**
069    * Returns the style table with the given name or <code>null</code> if
070    * there is none having the give name.
071    * 
072    * @throws IllegalArgumentException
073    *           if the name is null.
074    */
075   public static JavaSourceStyleTable getPredefinedTable(String name) {
076     if (name == null) {
077       throw new IllegalArgumentException("name is null."); //$NON-NLS-1$
078     }
079     JavaSourceStyleTable[] tables = getPredefinedTables();
080     for (int i = 0; i < tables.length; ++i) {
081       if (tables[i].getName().equalsIgnoreCase(name)) {
082         return tables[i];
083       }
084     }
085     return null;
086   }
