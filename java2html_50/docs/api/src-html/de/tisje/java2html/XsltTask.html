001 package de.tisje.java2html;
002 
003 import java.io.File;
004 import java.io.FileWriter;
005 import java.io.IOException;
006 import java.io.StringWriter;
007 
008 import de.java2html.converter.AbstractJavaSourceToXmlConverter;
009 import de.java2html.converter.JavaSource2XhtmlConverter;
010 import de.java2html.converter.JavaSource2XmlConverter;
011 import de.java2html.javasource.JavaSource;
012 import de.java2html.javasource.JavaSourceParser;
013 import de.java2html.options.JavaSourceConversionOptions;
014 import de.java2html.util.IoUtilities;
015 
016 /**
017  * This class is an interface between XSL and Java2Html.
018  * Before invoking, a namespace def must be added to the <code>xsl:stylesheet</code> tag:<br>
019  * <code>xmlns:j2h="de.tisje.java2html.XsltTask"</code><br>
020  * After that, it may be used this way:
021  * <pre>
022  *    &lt;xsl:value-of select="j2h:setSource(.)"/>
023  *    &lt;xsl:value-of select="j2h:writeFile('temp.xml')"/>
024  *    &lt;xsl:copy-of select="document('temp.xml')"/>
025  * </pre>
026  *
027  * @author <a href="mailto:Jan.Tisje@gmx.de">Jan Tisje</a>
028  * @version 1.0
029  */
030 //TODO Mar 11, 2004 (Markus Gebhard): This class urgently needs refactoring. Static instance variables - arg...
031 public class XsltTask {
032 
033   private static JavaSourceConversionOptions options = JavaSourceConversionOptions.getDefault();
034   private static AbstractJavaSourceToXmlConverter converter = new JavaSource2XhtmlConverter();
035   private static JavaSource source = null;
036   private static boolean xhtml = false;
037 
038   /**
039    * set options from xsl.
040    * @param lineNumbers if line numbers should be in the output code.
041    * @param pre if output code should be formatted using non-breaking spaces and &lt;br>.
042    * @param xhtml if output should be viewable stand-alone.
043    */
044   public static void setOptions(boolean lineNumbers, boolean pre, boolean xhtml) {
045     if (xhtml) {
046       converter = new JavaSource2XhtmlConverter();
047     }else {
048       converter = new JavaSource2XmlConverter();
049     }
050     converter.setOptions(lineNumbers, pre);
051     XsltTask.xhtml = xhtml;
052   }
053 
054   /** hand over java source read from main xml file. */
055   public static void setSource(String javaSource) {
056     source = new JavaSourceParser().parse(javaSource);
057   }
058 
059   /** read java source from file. */
060   public static void readFile(String javaFile) throws IOException {
061     source = new JavaSourceParser().parse(new File(javaFile));
062   }
063 
064   /** return java source in text form, html codes will be escaped. */
065   public static String getSource() throws IOException {
066     StringWriter writer = new StringWriter();
067     writer.write(converter.getDocumentHeader(options, "")); //$NON-NLS-1$
068     converter.convert(source, options, writer);
069     writer.write(converter.getDocumentFooter(options));
070     return writer.getBuffer().toString();
071   }
072 
073   /** output file to a separate xml file, less problems. 
074    * @deprecated As of Mar 11, 2004 (Markus Gebhard), replaced by {@link #writeFile(File)}*/
075   public static void writeFile(String filename) throws IOException {
076     writeFile(new File(filename));
077   }
078 
079   /** output file to a separate xml file */ 
080   public static void writeFile(File file) throws IOException {
081     FileWriter fileWriter = new FileWriter(file);
082     try {
083       fileWriter.write(converter.getDocumentHeader(options, "")); //$NON-NLS-1$
084       converter.convert(source, options, fileWriter);
085       fileWriter.write(converter.getDocumentFooter(options));
086     }
087     finally {
088       IoUtilities.close(fileWriter);
089     }
090   }
091 
092   /** use this class like a common comandline tool.
093    changing of options is not supported, yet */
094   public static void main(String args[]) {
095     String ext = ".xhtml"; //$NON-NLS-1$
096     if (!xhtml)
097       ext = ".xml"; //$NON-NLS-1$
098     for (int i = 0; args.length > i; i++) {
099       try {
100         readFile(args[i]);
101         writeFile(args[i] + ext);
102       }
103       catch (IOException e) {
104         e.printStackTrace();
105       }
106     }
107   }
108 }
